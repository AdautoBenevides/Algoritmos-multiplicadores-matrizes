#!/bin/bash

# Função para executar o código Java do servidor
run_java_server() {
    echo "Starting server..."
    java TCPServer
}

# Função para executar o código Java do cliente
run_java_client() {
    echo "Starting client..."
    java TCPClient
}

# Inicia o servidor Java em segundo plano
run_java_server &

# Espera um momento para o servidor iniciar completamente
sleep 2

# Executa dois clientes Java simultaneamente
run_java_client &
run_java_client &
